   550→  private buildHoop(pos: THREE.Vector3): THREE.Group {
   551→    const hoop = new THREE.Group();
   552→    
   553→    // Backboard
   554→    const backboardGeo = new THREE.BoxGeometry(1.8, 1.2, 0.05);
   555→    const backboardMat = new THREE.MeshStandardMaterial({
   556→      color: 0xffffff,

   559→      roughness: 0.1,
   560→      metalness: 0.3,
   561→    });
   562→    const backboard = new THREE.Mesh(backboardGeo, backboardMat);
   563→    backboard.position.set(0, 0.6, pos.z > 0 ? 0.8 : -0.8);
   564→    hoop.add(backboard);
   565→    
   566→    // Backboard frame
   567→    const frameMat = new THREE.MeshStandardMaterial({ color: 0x333333, metalness: 0.8, roughness: 0.2 });
   568→    const frameTop = new THREE.Mesh(new THREE.BoxGeometry(1.9, 0.05, 0.08), frameMat);
   569→    frameTop.position.copy(backboard.position);
   570→    frameTop.position.y += 0.625;
   571→    hoop.add(frameTop);
   572→    
   573→    // Pole
   574→    const poleGeo = new THREE.CylinderGeometry(0.08, 0.08, HOOP_HEIGHT + 1, 8);
   575→    const poleMat = new THREE.MeshStandardMaterial({ color: 0x444444, metalness: 0.8, roughness: 0.3 });
   576→    const pole = new THREE.Mesh(poleGeo, poleMat);
   577→    pole.position.set(0, (HOOP_HEIGHT + 1) / 2 - 0.5, pos.z > 0 ? 1.2 : -1.2);
   578→    pole.castShadow = true;
   579→    hoop.add(pole);
   580→    
   581→    // Rim
   582→    const rimGeo = new THREE.TorusGeometry(0.45, 0.02, 8, 24);
   583→    const rimMat = new THREE.MeshStandardMaterial({ color: 0xff4444, metalness: 0.9, roughness: 0.1 });
   584→    const rim = new THREE.Mesh(rimGeo, rimMat);
   585→    rim.position.set(0, HOOP_HEIGHT, pos.z > 0 ? 0 : 0);
   586→    rim.rotation.x = Math.PI / 2;
   587→    rim.castShadow = true;
   588→    hoop.add(rim);
   589→    
   590→    // Net (simplified with lines)
   591→    const netMat = new THREE.LineBasicMaterial({ color: 0xffffff, transparent: true, opacity: 0.6 });
   592→    for (let i = 0; i < 12; i++) {
   593→      const angle = (i / 12) * Math.PI * 2;
   594→      const x = Math.cos(angle) * 0.45;
   595→      const z = Math.sin(angle) * 0.45;
   596→      const netGeo = new THREE.BufferGeometry();
   597→      netGeo.setAttribute('position', new THREE.Float32BufferAttribute([
   598→        x, HOOP_HEIGHT, z,
   599→        Math.cos(angle) * 0.15, HOOP_HEIGHT - 0.5, Math.sin(angle) * 0.15,
   600→      ], 3));
   601→      hoop.add(new THREE.Line(netGeo, netMat));
   602→    }
   603→    
   604→    // Horizontal net rings
   605→    for (let j = 1; j <= 3; j++) {
   606→      const ringPoints: number[] = [];
   607→      const shrink = j / 4;
   608→      for (let i = 0; i <= 24; i++) {
   609→        const angle = (i / 24) * Math.PI * 2;
   610→        ringPoints.push(
   611→          Math.cos(angle) * 0.45 * (1 - shrink),
   612→          HOOP_HEIGHT - j * 0.12,
   613→          Math.sin(angle) * 0.45 * (1 - shrink)
   614→        );
   615→      }
   616→      const ringGeo = new THREE.BufferGeometry();
   617→      ringGeo.setAttribute('position', new THREE.Float32BufferAttribute(ringPoints, 3));
   618→      hoop.add(new THREE.Line(ringGeo, netMat));
   619→    }
   620→    
   621→    hoop.position.set(pos.x, 0, pos.z);
   622→    return hoop;
   623→  }

  1015→  private setupInput() {
  1016→    const onKeyDown = (e: KeyboardEvent) => {
  1017→      this.keys.add(e.code);
  1018→      if (e.code === 'Space') {
  1019→        e.preventDefault();
  1020→        if (!this.player.isJumping && !this.player.isShooting && !this.player.isDunking) {
  1021→          this.player.isJumping = true;
  1022→          this.player.jumpVelocity = JUMP_FORCE;
  1023→        }
  1024→      }
  1025→      if (e.code === 'KeyE') {
  1026→        e.preventDefault();
  1027→        if (this.hasBall === 'player' && !this.player.isShooting && !this.player.isDunking) {
  1028→          this.startPlayerShot();
  1029→        }
  1030→      }
  1031→      if (e.code === 'KeyQ') {
  1032→        e.preventDefault();
  1033→        this.trySteal('player');
  1034→      }
  1035→      if (e.code === 'KeyF') {
  1036→        e.preventDefault();
  1037→        this.tryDunk();
  1038→      }
  1039→      if (e.code === 'ShiftLeft' || e.code === 'ShiftRight') {
  1040→        this.keys.add('Sprint');
  1041→      }
  1042→    };

  1250→  // ============== PHYSICS ==============
  1251→  private updatePhysics(dt: number) {

  1258→    
  1259→    if (moveX !== 0 || moveZ !== 0) {
  1260→      const len = Math.sqrt(moveX * moveX + moveZ * moveZ);
  1261→      moveX /= len;
  1262→      moveZ /= len;
  1263→

  1265→      this.player.position.x += moveX * speed * dt;
  1266→      this.player.position.z += moveZ * speed * dt;
  1267→

  1270→

  1274→    
  1275→    // Court bounds for player
  1276→    this.player.position.x = THREE.MathUtils.clamp(this.player.position.x, -COURT_WIDTH / 2 + 1, COURT_WIDTH / 2 - 1);
  1277→    this.player.position.z = THREE.MathUtils.clamp(this.player.position.z, -COURT_LENGTH / 2 + 1, COURT_LENGTH / 2 - 1);
  1278→

  1280→    if (this.player.isJumping) {
  1281→      this.player.position.y += this.player.jumpVelocity * dt;
  1282→      this.player.jumpVelocity += GRAVITY * dt;

  1284→        this.player.position.y = 0;
  1285→        this.player.isJumping = false;
  1286→        this.player.jumpVelocity = 0;
  1287→        
  1288→        if (this.player.isDunking) {
  1289→          this.player.isDunking = false;
  1290→          this.checkDunkScore('player');
  1291→        }

  1294→    
  1295→    // Jump physics - Opponent
  1296→    if (this.opponent.isJumping) {
  1297→      this.opponent.position.y += this.opponent.jumpVelocity * dt;
  1298→      this.opponent.jumpVelocity += GRAVITY * dt;
  1299→      if (this.opponent.position.y <= 0) {
  1300→        this.opponent.position.y = 0;
  1301→        this.opponent.isJumping = false;
  1302→        this.opponent.jumpVelocity = 0;
  1303→        
  1304→        if (this.opponent.isDunking) {
  1305→          this.opponent.isDunking = false;
  1306→          this.checkDunkScore('opponent');
  1307→        }
  1308→      }
  1309→    }
  1310→    
  1311→    // Shot power charge
  1312→    if (this.isShootingHold) {
  1313→      this.shootPower = Math.min(SHOOT_POWER_MAX, this.shootPower + dt * 20);
  1314→    }
  1315→    
  1316→    // Ball physics
  1317→    if (this.ball.isAirborne) {
  1318→      this.ball.velocity.y += GRAVITY * dt;
  1319→      this.ball.position.add(this.ball.velocity.clone().multiplyScalar(dt));
  1320→      this.ball.rotation.x += dt * 5;
  1321→      this.ball.rotation.z += dt * 3;
  1322→      
  1323→      // Check hoop collisions
  1324→      this.checkHoopCollision();
  1325→      
  1326→      // Ground bounce
  1327→      if (this.ball.position.y < BALL_RADIUS) {
  1328→        this.ball.position.y = BALL_RADIUS;
  1329→        this.ball.velocity.y = Math.abs(this.ball.velocity.y) * 0.5;
  1330→        this.ball.velocity.x *= 0.8;
  1331→        this.ball.velocity.z *= 0.8;
  1332→        
  1333→        // If ball is slow enough and on ground, it's a rebound
  1334→        if (this.ball.velocity.length() < 1) {
  1335→          this.ball.isAirborne = false;
  1336→          // Who picks it up?
  1337→          const distToPlayer = this.ball.position.distanceTo(this.player.position);
  1338→          const distToOpponent = this.ball.position.distanceTo(this.opponent.position);
  1339→          if (distToPlayer < distToOpponent) {
  1340→            this.hasBall = 'player';
  1341→            this.ball.owner = 'player';
  1342→          } else {
  1343→            this.hasBall = 'opponent';
  1344→            this.ball.owner = 'opponent';
  1345→          }
  1346→        }
  1347→      }
  1348→      
  1349→      // Court bounds for ball
  1350→      if (Math.abs(this.ball.position.x) > COURT_WIDTH / 2 + 2) {
  1351→        this.ball.velocity.x *= -0.5;
  1352→        this.ball.position.x = THREE.MathUtils.clamp(this.ball.position.x, -COURT_WIDTH / 2 + 1, COURT_WIDTH / 2 - 1);
  1353→      }
  1354→      if (Math.abs(this.ball.position.z) > COURT_LENGTH / 2 + 2) {
  1355→        this.ball.velocity.z *= -0.5;
  1356→        this.ball.position.z = THREE.MathUtils.clamp(this.ball.position.z, -COURT_LENGTH / 2 + 1, COURT_LENGTH / 2 - 1);
  1357→      }
  1358→    } else {
  1359→      // Ball follows owner
  1360→      if (this.ball.owner === 'player') {
  1361→        const dribbleY = 0.8 + Math.abs(Math.sin(this.dribblePhase * 2)) * 0.5;
  1362→        this.ball.position.set(
  1363→          this.player.position.x + Math.sin(this.player.rotation) * 0.5,
  1364→          dribbleY,
  1365→          this.player.position.z + Math.cos(this.player.rotation) * 0.5
  1366→        );
  1367→      } else if (this.ball.owner === 'opponent') {

  1369→        this.ball.position.set(
  1370→          this.opponent.position.x + Math.sin(this.opponent.rotation) * 0.5,
  1371→          oppDribble,
  1372→          this.opponent.position.z + Math.cos(this.opponent.rotation) * 0.5
  1373→        );
  1374→      }
  1375→    }
  1376→    
  1377→    // Cooldowns
  1378→    if (this.stealCooldown > 0) this.stealCooldown -= dt;
  1379→    if (this.scoreCooldown > 0) this.scoreCooldown -= dt;
  1380→    
  1381→    // Player-Opponent collision
  1382→    const pDist = this.player.position.distanceTo(this.opponent.position);
  1383→    if (pDist < PLAYER_RADIUS * 2.2) {
  1384→      const pushDir = new THREE.Vector3()
  1385→        .subVectors(this.player.position, this.opponent.position)

  1389→      this.opponent.position.sub(pushDir.clone().multiplyScalar(overlap));
  1390→    }
  1391→  }

  1532→    for (let i = 0; i < this.particles.length; i++) {
  1533→      const p = this.particles[i];
  1534→      if (p.life <= 0) {
  1535→        positions.setXYZ(i, 0, -100, 0);
  1536→        continue;
  1537→      }
  1538→      
  1539→      p.position.add(p.velocity.clone().multiplyScalar(dt));
  1540→      p.velocity.y += GRAVITY * 0.5 * dt;
  1541→      p.life -= dt / p.maxLife;
  1542→      
  1543→      positions.setXYZ(i, p.position.x, p.position.y, p.position.z);
  1544→      colors.setXYZ(i, p.color.r * p.life, p.color.g * p.life, p.color.b * p.life);
  1545→    }
  1546→    
  1547→    positions.needsUpdate = true;
  1548→    colors.needsUpdate = true;
  1549→  }

  1551→  // ============== CAMERA ==============
  1552→  private cameraShake(intensity: number, duration: number) {
  1553→    const startTime = this.clock.elapsedTime;
  1554→    const shake = () => {
  1555→      const elapsed = this.clock.elapsedTime - startTime;
  1556→      if (elapsed > duration) return;
  1557→      const decay = 1 - (elapsed / duration);
  1558→      this.camera.position.x += (Math.random() - 0.5) * intensity * decay;
  1559→      this.camera.position.y += (Math.random() - 0.5) * intensity * decay * 0.5;
  1560→      requestAnimationFrame(shake);
  1561→    };
  1562→    shake();
  1563→  }

  1565→  private updateCamera() {
  1566→    // Camera follows the action - centered between players
  1567→    const midX = (this.player.position.x + this.opponent.position.x) / 2;
  1568→    const midZ = (this.player.position.z + this.opponent.position.z) / 2;
  1569→    
  1570→    // If ball is in the air, look at ball
  1571→    let lookTarget: THREE.Vector3;
  1572→    if (this.ball.isAirborne) {
  1573→      lookTarget = this.ball.position.clone();
  1574→    } else {
  1575→      lookTarget = new THREE.Vector3(midX, 1.5, midZ);
  1576→    }
  1577→    
  1578→    // Smooth camera movement
  1579→    const targetCamPos = new THREE.Vector3(
  1580→      midX * 0.3,
  1581→      10 + Math.abs(this.player.position.z - this.opponent.position.z) * 0.15,
  1582→      midZ + 16
  1583→    );
  1584→    
  1585→    this.camera.position.lerp(targetCamPos, 0.03);
  1586→    
  1587→    const currentLook = new THREE.Vector3();
  1588→    this.camera.getWorldDirection(currentLook);
  1589→    this.camera.lookAt(lookTarget);
  1590→  }

  1593→  private updateAnimations(dt: number) {
  1594→    // Player running animation
  1595→    if (this.keys.has('KeyA') || this.keys.has('KeyD') || this.keys.has('KeyW') || this.keys.has('KeyS') ||
  1596→        this.keys.has('ArrowLeft') || this.keys.has('ArrowRight') || this.keys.has('ArrowUp') || this.keys.has('ArrowDown')) {
  1597→      this.player.animTimer += dt * 8;
  1598→      const swing = Math.sin(this.player.animTimer) * 0.4;
  1599→      const leftArm = this.playerMesh.getObjectByName('leftArm');
  1600→      const rightArm = this.playerMesh.getObjectByName('rightArm');
  1601→      const leftLeg = this.playerMesh.getObjectByName('leftLeg');
  1602→      const rightLeg = this.playerMesh.getObjectByName('rightLeg');
  1603→      if (leftArm) leftArm.rotation.x = swing;
  1604→      if (rightArm) rightArm.rotation.x = -swing;
  1605→      if (leftLeg) leftLeg.rotation.x = -swing * 0.6;
  1606→      if (rightLeg) rightLeg.rotation.x = swing * 0.6;
  1607→    } else {
  1608→      // Reset to idle
  1609→      const leftArm = this.playerMesh.getObjectByName('leftArm');
  1610→      const rightArm = this.playerMesh.getObjectByName('rightArm');
  1611→      const leftLeg = this.playerMesh.getObjectByName('leftLeg');
  1612→      const rightLeg = this.playerMesh.getObjectByName('rightLeg');
  1613→      if (leftArm) leftArm.rotation.x *= 0.9;
  1614→      if (rightArm) rightArm.rotation.x *= 0.9;
  1615→      if (leftLeg) leftLeg.rotation.x *= 0.9;
  1616→      if (rightLeg) rightLeg.rotation.x *= 0.9;
  1617→    }
  1618→    
  1619→    // Shooting animation
  1620→    if (this.player.isShooting) {
  1621→      const rightArm = this.playerMesh.getObjectByName('rightArm');
  1622→      if (rightArm) rightArm.rotation.x = -Math.PI * 0.8;
  1623→    }
  1624→    
  1625→    // Dunk animation
  1626→    if (this.player.isDunking && this.player.isJumping) {
  1627→      const rightArm = this.playerMesh.getObjectByName('rightArm');
  1628→      if (rightArm) {
  1629→        rightArm.rotation.x = -Math.PI * 0.95;
  1630→        rightArm.rotation.z = -0.5;
  1631→      }
  1632→    }
  1633→    
  1634→    // Opponent running animation
  1635→    this.opponent.animTimer += dt * 6;
  1636→    const oppSwing = Math.sin(this.opponent.animTimer) * 0.3;
  1637→    const oppLeftArm = this.opponentMesh.getObjectByName('leftArm');
  1638→    const oppRightArm = this.opponentMesh.getObjectByName('rightArm');
  1639→    const oppLeftLeg = this.opponentMesh.getObjectByName('leftLeg');
  1640→    const oppRightLeg = this.opponentMesh.getObjectByName('rightLeg');
  1641→    if (oppLeftArm) oppLeftArm.rotation.x = oppSwing;
  1642→    if (oppRightArm) oppRightArm.rotation.x = -oppSwing;
  1643→    if (oppLeftLeg) oppLeftLeg.rotation.x = -oppSwing * 0.6;
  1644→    if (oppRightLeg) oppRightLeg.rotation.x = oppSwing * 0.6;
  1645→    
  1646→    // Update mesh positions
  1647→    this.playerMesh.position.copy(this.player.position);
  1648→    this.playerMesh.rotation.y = this.player.rotation;
  1649→    
  1650→    this.opponentMesh.position.copy(this.opponent.position);
  1651→    this.opponentMesh.rotation.y = this.opponent.rotation;
  1652→    
  1653→    // Ball mesh
  1654→    this.ballMesh.position.copy(this.ball.position);
  1655→    this.ballMesh.rotation.copy(this.ball.rotation);
  1656→    
  1657→    // Update ball trail
  1658→    this.updateBallTrail();
  1659→  }
